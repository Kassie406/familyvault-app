import React, { useState, useCallback, useRef } from 'react';
import { motion, AnimatePresence } from 'framer-motion';

// Upload States
const UPLOAD_STATES = {
  EMPTY: 'empty',
  DRAGOVER: 'dragover',
  UPLOADING: 'uploading',
  PROCESSING: 'processing',
  COMPLETE: 'complete',
  ERROR: 'error'
};

// Main Upload Center Component
export const EnhancedUploadCenter = () => {
  const [uploadState, setUploadState] = useState(UPLOAD_STATES.EMPTY);
  const [uploadProgress, setUploadProgress] = useState(0);
  const [analysisResults, setAnalysisResults] = useState(null);
  const [currentFile, setCurrentFile] = useState(null);
  const [isDragActive, setIsDragActive] = useState(false);
  const fileInputRef = useRef(null);

  // File upload handler
  const handleFileUpload = useCallback(async (files) => {
    const file = files[0];
    if (!file) return;

    setCurrentFile(file);
    setUploadState(UPLOAD_STATES.UPLOADING);
    setUploadProgress(0);

    try {
      // Simulate upload progress
      await simulateUploadProgress();
      
      // Start AI analysis
      setUploadState(UPLOAD_STATES.PROCESSING);
      const results = await analyzeDocumentWithAI(file);
      
      setAnalysisResults(results);
      setUploadState(UPLOAD_STATES.COMPLETE);
    } catch (error) {
      console.error('Upload failed:', error);
      setUploadState(UPLOAD_STATES.ERROR);
    }
  }, []);

  // Drag and drop handlers
  const handleDragEnter = useCallback((e) => {
    e.preventDefault();
    e.stopPropagation();
    setIsDragActive(true);
    if (uploadState === UPLOAD_STATES.EMPTY) {
      setUploadState(UPLOAD_STATES.DRAGOVER);
    }
  }, [uploadState]);

  const handleDragLeave = useCallback((e) => {
    e.preventDefault();
    e.stopPropagation();
    setIsDragActive(false);
    if (uploadState === UPLOAD_STATES.DRAGOVER) {
      setUploadState(UPLOAD_STATES.EMPTY);
    }
  }, [uploadState]);

  const handleDragOver = useCallback((e) => {
    e.preventDefault();
    e.stopPropagation();
  }, []);

  const handleDrop = useCallback((e) => {
    e.preventDefault();
    e.stopPropagation();
    setIsDragActive(false);
    
    const files = Array.from(e.dataTransfer.files);
    if (files.length > 0) {
      handleFileUpload(files);
    }
  }, [handleFileUpload]);

  // File input click handler
  const handleFileInputClick = () => {
    fileInputRef.current?.click();
  };

  const handleFileInputChange = (e) => {
    const files = Array.from(e.target.files || []);
    if (files.length > 0) {
      handleFileUpload(files);
    }
  };

  // Reset upload state
  const resetUpload = () => {
    setUploadState(UPLOAD_STATES.EMPTY);
    setUploadProgress(0);
    setAnalysisResults(null);
    setCurrentFile(null);
  };

  return (
    <div className="enhanced-upload-center">
      <div className="upload-main-area">
        <motion.div
          className={`upload-zone ${uploadState} ${isDragActive ? 'drag-active' : ''}`}
          onDragEnter={handleDragEnter}
          onDragLeave={handleDragLeave}
          onDragOver={handleDragOver}
          onDrop={handleDrop}
          onClick={uploadState === UPLOAD_STATES.EMPTY ? handleFileInputClick : undefined}
          whileHover={uploadState === UPLOAD_STATES.EMPTY ? { y: -2 } : {}}
          transition={{ duration: 0.2 }}
        >
          <input
            ref={fileInputRef}
            type="file"
            onChange={handleFileInputChange}
            accept=".pdf,.jpg,.jpeg,.png,.doc,.docx"
            style={{ display: 'none' }}
          />
          
          <StateRenderer
            state={uploadState}
            progress={uploadProgress}
            file={currentFile}
            results={analysisResults}
            onReset={resetUpload}
          />
        </motion.div>
      </div>

      <div className="upload-sidebar">
        <RecentUploads />
        <QuickActions />
      </div>
    </div>
  );
};

// State Renderer Component
const StateRenderer = ({ state, progress, file, results, onReset }) => {
  return (
    <AnimatePresence mode="wait">
      {state === UPLOAD_STATES.EMPTY && <EmptyState key="empty" />}
      {state === UPLOAD_STATES.DRAGOVER && <DragOverState key="dragover" />}
      {state === UPLOAD_STATES.UPLOADING && <UploadingState key="uploading" progress={progress} file={file} />}
      {state === UPLOAD_STATES.PROCESSING && <ProcessingState key="processing" file={file} />}
      {state === UPLOAD_STATES.COMPLETE && <CompleteState key="complete" results={results} onReset={onReset} />}
      {state === UPLOAD_STATES.ERROR && <ErrorState key="error" onReset={onReset} />}
    </AnimatePresence>
  );
};

  >

        />
      </motion.svg>
    </div>
    
    <h3 className="upload-title">Upload Family Documents</h3>
    <p className="upload-description">
      Drag & drop files here or click to browse

    </div>
  </motion.div>

        <ProcessingStep label="Identifying family members" />
      </div>

);

  <motion.div
    className="upload-state upload-complete-state"
    initial={{ opacity: 0, scale: 0.9 }}

        Upload Another
      </motion.button>
    </div>
  </motion.div>
);

// Utility Functions
const simulateUploadProgress = () => {
  return new Promise((resolve) => {
    let progress = 0;
    const interval = setInterval(() => {
      progress += Math.random() * 30;
      if (progress >= 100) {
        progress = 100;
        clearInterval(interval);
        resolve();
      }
    }, 200);
  });
};

const analyzeDocumentWithAI = async (file) => {
  // Simulate AI analysis delay
  await new Promise(resolve => setTimeout(resolve, 3000));
  
  // Mock analysis results based on existing data format
  return {
    extractedFields: [
      { key: "Issuer", value: "Social Security Administration", confidence: 85 },
      { key: "Document Type", value: "Driver License", confidence: 92 },
      { key: "Name", value: "Angel Quintana", confidence: 88 }
    ],
    documentType: "Identity Document",
    confidence: 88
  };
};

const formatFileSize = (bytes) => {
  if (!bytes) return '0 B';
  const k = 1024;
  const sizes = ['B', 'KB', 'MB', 'GB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
};
