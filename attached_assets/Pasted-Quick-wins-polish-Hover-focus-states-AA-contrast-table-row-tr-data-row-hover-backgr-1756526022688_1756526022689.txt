Tenant filter menu (counts + no role leakage)

// assumes <tr data-tenant="PUBLIC|FAMILY|STAFF">
const rows=[...document.querySelectorAll('tbody tr[data-tenant]')];
const counts=rows.reduce((a,r)=>((a[r.dataset.tenant]=(a[r.dataset.tenant]||0)+1),a),{});
['ALL','PUBLIC','FAMILY','STAFF'].forEach(k=>{
  const el=document.querySelector(`.tenant-menu [data-tenant="${k}"]`);
  if(el){ el.querySelector('.count').textContent = k==='ALL' ? rows.length : (counts[k]||0); }
});
