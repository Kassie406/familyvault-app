        }}
      />
      
      {/* Your existing sidebar component */}
      {sidebarOpen && (
        <DocumentSidebar 
          documents={documents}
          onClose={() => setSidebarOpen(false)}
        />
      )}
    </div>
  );
};

export default FamilyInboxPanel;
