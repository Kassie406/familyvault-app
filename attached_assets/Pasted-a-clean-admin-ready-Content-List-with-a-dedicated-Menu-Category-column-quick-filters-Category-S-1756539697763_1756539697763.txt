JS — client-side filters
<script>
(function(){
  const rows = [...document.querySelectorAll('#articles-table tbody tr')];
  const fTenant = document.getElementById('f-tenant');
  const fCat    = document.getElementById('f-category');
  const fStatus = document.getElementById('f-status');
  const fSearch = document.getElementById('f-search');
  const fClear  = document.getElementById('f-clear');

  function matches(r){
    const tOK = fTenant.value==='ALL' || r.dataset.tenant===fTenant.value;
    const cOK = fCat.value==='ALL'    || r.dataset.category===fCat.value;
    const sOK = fStatus.value==='ALL' || r.dataset.status===fStatus.value;
    const q   = fSearch.value.trim().toLowerCase();
    const qOK = !q || (r.dataset.title||'').toLowerCase().includes(q) ||
                      (r.dataset.author||'').toLowerCase().includes(q);
    return tOK && cOK && sOK && qOK;
  }

  function apply(){
    let shown=0;
    rows.forEach(r=>{ const show = matches(r); r.hidden=!show; if(show) shown++; });
    // optional: empty state
    let empty = document.querySelector('#articles-table tbody tr.empty-row');
    if(!empty){
      empty = document.createElement('tr'); empty.className='empty-row'; empty.hidden=true;
      empty.innerHTML = `<td colspan="7" class="tc muted" style="padding:16px">No articles match.</td>`;
      document.querySelector('#articles-table tbody').appendChild(empty);
    }
    empty.hidden = shown!==0;
  }

  [fTenant,fCat,fStatus,fSearch].forEach(el=>el.addEventListener('input', apply));
  fClear.addEventListener('click', ()=>{ fTenant.value='ALL'; fCat.value='ALL'; fStatus.value='ALL'; fSearch.value=''; apply(); });
  apply();
})();
</script>

Why this helps

Menu Category column lets you (and other admins) see at a glance which mega-menu section an article powers.

Filters mirror your three-audience model (Public / Family / Staff) and editorial states (Draft / Scheduled / Published).

Consistent actions keep usability high: Edit / Preview / Publish-state toggle / Archive.
