Minimal JS (scope tabs + Stripe empty-state demo)
<script>
  // Toggle segment scope (visual only; FAMILY/STAFF show info-only banner)
  document.querySelectorAll('.segmented .seg').forEach(b=>{
    b.addEventListener('click', ()=>{
      document.querySelectorAll('.seg').forEach(x=>x.classList.remove('is-active'));
      b.classList.add('is-active');
      const scope=b.dataset.scope;
      const table=document.getElementById('plans-table');
      // For this admin view, only PUBLIC has billable rows:
      table.parentElement.style.display = scope==='PUBLIC' ? '' : 'none';
      document.getElementById('stripe-empty').hidden = (scope!=='PUBLIC'); // FAMILY/STAFF: no billing
      if(scope!=='PUBLIC'){
        document.getElementById('stripe-empty').querySelector('h3').textContent =
          scope==='FAMILY' ? 'Family plans do not use billing' : 'Staff plans do not use billing';
        document.getElementById('stripe-empty').querySelector('p').textContent =
          'Access is role-based only. Manage roles and permissions in Users.';
        document.getElementById('btn-connect-stripe').hidden = true;
      } else {
        document.getElementById('btn-connect-stripe').hidden = false;
      }
    });
  });

  // Example: If Stripe is not connected, show empty-state
  const stripeConnected = true; // hook to your real flag
  document.getElementById('stripe-empty').hidden = stripeConnected;
</script>
